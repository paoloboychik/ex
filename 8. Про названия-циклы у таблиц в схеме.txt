/*создаём таблицы с подходящими для тестирования названиями*/
CREATE TABLE BAA(NUM NUMBER(2));
CREATE TABLE ABA(NUM NUMBER(2));
CREATE TABLE AAB(NUM NUMBER(2));

/*получаем все имена таблиц, которыми владеет пользователь, из представления user_tables*/
WITH tables AS (
SELECT table_name
FROM user_tables )

SELECT ROWNUM AS "Номер", t1.table_name AS "Таблица 1", t2.table_name AS "Таблица 2"

/*используем внутреннее соединение (INNER JOIN), чтобы в результат попали только соответствующие заданному условию строки,
другими словами, в результате не будет строк, которые относятся только к одной из таблиц, т е строк, не прошедших условие*/
FROM tables t1 INNER JOIN tables t2 
/*условие соединения - в первую очередь одинаковая длина названий
(только при таком условии одно название может получиться с помощью циклического сдвига другого)*/
ON (LENGTH(t1.table_name) = LENGTH(t2.table_name))
/*далее с помощью оператора WHERE отбираем такие строки,
в которых названия таблиц разные и в которых название одной таблицы является подстрокой в "удвоенном" варианте названия второй таблицы*/
WHERE T1.table_name != T2.table_name AND
INSTR(T2.table_name||T2.table_name, T1.table_name) != 0;
